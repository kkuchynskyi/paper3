\begin{abstract}
    We study interacting SDEs driven by common noise where the diffusion coefficient is state-dependent.
    We derive the SDE for the Jacobian matrix of the associated stochastic flow and obtain a pathwise exponential bound on directional derivatives, which yields a pathwise logarithmic Sobolev inequality for the transported measures with a random constant controlled by the flow Lipschitz constant.
\end{abstract}

\section{Introduction}\label{sec:intro}

We consider Stochastic Differential Equations with Interactions that were proposed by Andrey A. Dorogovtsev in \cite{dorogovtsev2003},
% PDF landmark: base interacting SDE with common noise.
\begin{equation}\label{eq:main_eqn}
    \begin{cases}
        dX_t(u) = a\big(X_t(u),\mu_t\big)\,dt + \sigma\big(X_t(u)\big)\,dB_t, \\[2mm]
        X_0(u)=u, u \in \mathbb{R}^d                                          \\[1mm]
        \mu_t = \mu_0\circ X_t^{-1}.
    \end{cases}
\end{equation}
Here \(\mu_0\) is a deterministic initial probability measure,
\(a:\mathbb{R}^d\times\mathcal{P}_2(\mathbb{R}^d)\to\mathbb{R}^d\) is the drift, where
\(\mathcal{P}_2(\mathbb{R}^d)\) denotes the space of Borel probability measures on \(\mathbb{R}^d\) with finite second moment,
and the diffusion coefficient \(\sigma:\mathbb{R}^d\to\mathbb{R}^{d\times d}\) is state-dependent and non-constant.

In this paper we will consider specific form of the sdewi with the dependence on \( \mu_t \)
remains explicit through the kernel \( K \) and the order parameter \( \Theta_t \).
% PDF landmark: kernel/order-parameter form of the SDEWI.
\begin{equation}\label{eq:kernel-sdewi}
    \left\{
    \begin{aligned}
        dX_t(u) & =
        \bigg[ b(X_t(u),\Theta_t) + \int_{\mathbb{R}^d} K(X_t(u),X_t(v),\Theta_t)\,\mu_0(dv) \bigg]dt \\
                & \quad+ \sum_{\ell=1}^m \sigma_\ell(X_t(u),\Theta_t)\,dB_t^\ell,                     \\
        X_0(u)  & =u,\qquad u\in\mathbb{R}^d,                                                         \\
        \mu_t   & =\mu_0\circ X_t^{-1}.
    \end{aligned}
    \right.
\end{equation}
where
\begin{equation*}
    \theta(\mu):=\int_{\mathbb{R}^d}\Phi(x)\,\mu(dx),
\end{equation*}

% PDF landmark: Assumption on global Lipschitz and growth hypotheses.
\begin{assumption}[Global Lipschitz and growth hypotheses]\label{ass:global-lip}
    The maps \(b,K,\sigma_\ell,\Phi\) are globally Lipschitz and have at most linear growth.
    More explicitly, there exists \(L<\infty\) such that
    \begin{align*}
        |\Phi(x)-\Phi(y)|           & \le L|x-y|,                               \\
        |b(x,\theta)-b(y,\eta)|     & \le L\big(|x-y|+|\theta-\eta|\big),       \\
        |K(x,z,\theta)-K(y,w,\eta)| & \le L\big(|x-y|+|z-w|+|\theta-\eta|\big),
    \end{align*}
    and
    \begin{equation*}
        \sum_{\ell=1}^m |\sigma_\ell(x,\theta)-\sigma_\ell(y,\eta)|^2 \le L^2\big(|x-y|^2+|\theta-\eta|^2\big).
    \end{equation*}
\end{assumption}

\subsection{Dissipativity assumptions}
% Local comment: pairwise energy measures spread between two labels.
Define the pairwise energy
\begin{equation*}
    \mathcal D_t := \iint_{\mathbb{R}^d\times\mathbb{R}^d} |X_t(u)-X_t(v)|^2\,\mu_0(du)\mu_0(dv).
\end{equation*}

Equivalently, if
\begin{equation*}
    \bar X_t:=\int_{\mathbb{R}^d}X_t(u)\,\mu_0(du),
\end{equation*}
and
\begin{equation*}
    \mathcal V_t := \int_{\mathbb{R}^d}|X_t(u)-\bar X_t|^2\,\mu_0(du),
\end{equation*}
then
\begin{equation*}
    \mathcal D_t=2\mathcal V_t,
\end{equation*}
Indeed,
\begin{align*}
    \mathcal D_t & = \iint |X_t(u)-X_t(v)|^2\,\mu_0(du)\mu_0(dv)                                                \\
                 & = \iint \big( |X_t(u)|^2+|X_t(v)|^2 -2\langle X_t(u),X_t(v)\rangle \big)\,\mu_0(du)\mu_0(dv) \\
                 & = 2\int |X_t(u)|^2\,\mu_0(du)- 2\left|\int X_t(u)\,\mu_0(du)\right|^2                        \\
                 & = 2\int |X_t(u)-\bar X_t|^2\,\mu_0(du)= 2\mathcal V_t.
\end{align*}
We assume the following dissipativity condition.
% PDF landmark: Assumption on pairwise dissipativity.
\begin{assumption}[Pairwise dissipativity]\label{ass:pairwise-dissipativity}
    There exists \(\lambda>0\) such that for every \(x,y\in\mathbb{R}^d\), every \(\theta\in\mathbb{R}^q\), and every
    \(\nu\in\mathcal P_2(\mathbb{R}^d)\),
    \begin{align*}
         & 2\left\langle x-y, b(x,\theta)-b(y,\theta) + \int_{\mathbb{R}^d} \big[K(x,z,\theta)-K(y,z,\theta)\big] \nu(dz) \right\rangle \\
         & \qquad + \sum_{\ell=1}^m |\sigma_\ell(x,\theta)-\sigma_\ell(y,\theta)|^2 \le -\lambda |x-y|^2.
    \end{align*}
\end{assumption}

The dissipativity condition covers several basic models.

\begin{example}[Mean-field Ornstein--Uhlenbeck interaction]
    Consider
    \begin{equation*}
        dX_t(u)= -\lambda_0\big(X_t(u)-m_t\big)\,dt + \sigma(X_t(u))\,dB_t, \qquad m_t=\int_{\mathbb{R}^d} X_t(v)\,\mu_0(dv).
    \end{equation*}
    If
    \begin{equation*}
        \|\sigma(x)-\sigma(y)\|_{\mathrm{HS}} \le L_\sigma |x-y|,
    \end{equation*}
    then
    \begin{equation*}
        \mathbb E\mathcal D_t \le e^{-(2\lambda_0-L_\sigma^2)t}\mathcal D_0.
    \end{equation*}
    Thus the cloud contracts whenever \(2\lambda_0>L_\sigma^2\).
\end{example}

\begin{example}[Potential interaction]
    Take
    \begin{equation*}
        b(x,\theta)=-\nabla V(x), \qquad K(x,y,\theta)=-\nabla W(x-y).
    \end{equation*}
    Then
    \begin{equation*}
        dX_t(u) =  \left[ -\nabla V(X_t(u)) - \int_{\mathbb{R}^d} \nabla W(X_t(u)-X_t(v))\,\mu_0(dv) \right]dt + \sigma(X_t(u))\,dB_t.
    \end{equation*}
    If
    \begin{equation*}
        D^2V\ge \rho I, \qquad D^2W\ge \kappa I, \qquad \|\sigma(x)-\sigma(y)\|_{\mathrm{HS}} \le L_\sigma |x-y|,
    \end{equation*}
    then
    \begin{equation*}
        \mathbb E\mathcal D_t \le e^{-(2\rho+2\kappa-L_\sigma^2)t}\mathcal D_0.
    \end{equation*}
\end{example}

\begin{example}[Nonlinear consensus toward an order parameter]
    Let
    \begin{equation*}
        \Theta_t=\int_{\mathbb{R}^d} \psi(X_t(v))\,\mu_0(dv),
    \end{equation*}
    and consider
    \begin{equation*}
        dX_t(u)
        =
        -\lambda_0\big(X_t(u)-\Theta_t\big)\,dt
        +
        \sigma(X_t(u),\Theta_t)\,dB_t.
    \end{equation*}
    For two labels \(u,v\), the order parameter cancels in the drift
    difference:
    \begin{equation*}
        -\lambda_0\big(X_t(u)-\Theta_t\big)
        +
        \lambda_0\big(X_t(v)-\Theta_t\big)
        =
        -\lambda_0\big(X_t(u)-X_t(v)\big).
    \end{equation*}
    Thus one obtains contraction under \(2\lambda_0>L_\sigma^2\). The
    order parameter may be nonlinear, for example
    \begin{equation*}
        \Theta_t=\int_{\mathbb{R}^d} \tanh(X_t(v))\,\mu_0(dv).
    \end{equation*}
\end{example}


\subsection{Proof}
% PDF landmark: Theorem on exponential synchronization in expected W2 spread.
\begin{theorem}[Synchronization under dissipativity]\label{thm:synchronization-dissipativity}
    Assume that \(\mu_0\in\mathcal P_2(\mathbb{R}^d)\) and that Assumption~\ref{ass:pairwise-dissipativity} holds. Then
    \begin{equation*}
        \mathbb E\mathcal D_t \le e^{-\lambda t}\mathcal D_0.
    \end{equation*}

    In particular,
    \begin{equation*}
        \mathbb E W_2^2(\mu_t,\delta_{\bar X_t}) \le e^{-\lambda t}W_2^2(\mu_0,\delta_{\bar X_0}).
    \end{equation*}
    Thus the transported measure \(\mu_t\) conventrates around its random center of mass \(\bar X_t\).
\end{theorem}

% Proof idea:
% 1. Differentiate the stochastic flow.
\begin{proof}
    % Proof roadmap: first prove pairwise contraction, then integrate over labels.
    Fix two labels \(u,v\in\mathbb{R}^d\). Define the differnce
    \begin{equation*}
        Z_t^{u,v}:=X_t(u)-X_t(v).
    \end{equation*}
    We first prove that under Assumption~\ref{ass:pairwise-dissipativity}
    \begin{equation*}
        \mathbb E|Z_t^{u,v}|^2 \le e^{-\lambda t}|u-v|^2,
    \end{equation*}
    then we integrate this estimate with respect to \(\mu_0(du)\mu_0(dv)\).

    Writing SDEWI \eqref{eq:kernel-sdewi} for \(X_t(u)\) and for \(X_t(v)\) and then substracting the two equations gives
    \begin{equation*}
        dZ_t^{u,v} = G_t^{u,v}\,dt + \sum_{\ell=1}^m H_{\ell,t}^{u,v}\,dB_t^\ell,
    \end{equation*}
    where theg drift coefficient
    \begin{align*}
        G_t^{u,v} & := b(X_t(u),\Theta_t)-b(X_t(v),\Theta_t)                                                                 \\
                  & \quad+  \int_{\mathbb{R}^d} \big[ K(X_t(u),X_t(w),\Theta_t) - K(X_t(v),X_t(w),\Theta_t)\big]\,\mu_0(dw),
    \end{align*}
    and teh diffusion coefficient is
    \begin{equation*}
        H_{\ell,t}^{u,v} := \sigma_\ell(X_t(u),\Theta_t) - \sigma_\ell(X_t(v),\Theta_t).
    \end{equation*}

    Since \(\mu_t =\mu_0\circ X_t^{-1}\), we may rewrite the interaction term as
    \begin{equation*}
        \int_{\mathbb{R}^d}K(X_t(u),X_t(w),\Theta_t)\,\mu_0(dw) = \int_{\mathbb{R}^d}K(X_t(u),z,\Theta_t)\,\mu_t(dz).
    \end{equation*}
    Thus
    \begin{align*}
        G_t^{u,v} & := b(X_t(u),\Theta_t)-b(X_t(v),\Theta_t)                                                       \\
                  & \quad+ \int_{\mathbb{R}^d} \big[ K(X_t(u),z,\Theta_t) - K(X_t(v),z,\Theta_t) \big]\,\mu_t(dz).
    \end{align*}

    Applying It\^o's formula to \(|Z_t^{u,v}|^2\), we obtain
    \begin{align*}
        d|Z_t^{u,v}|^2 & = 2\langle Z_t^{u,v},G_t^{u,v}\rangle\,dt + \sum_{\ell=1}^m |H_{\ell,t}^{u,v}|^2\,dt \\
                       & \quad+ 2\sum_{\ell=1}^m \langle Z_t^{u,v},H_{\ell,t}^{u,v}\rangle\,dB_t^\ell.
    \end{align*}
    Apply Assumption~\ref{ass:pairwise-dissipativity} with
    \(x=X_t(u)\), \(y=X_t(v)\), \(\theta=\Theta_t\), and \(\nu=\mu_t\). Then
    \begin{equation*} 2\langle Z_t^{u,v},G_t^{u,v}\rangle +  \sum_{\ell=1}^m |H_{\ell,t}^{u,v}|^2 \le -\lambda |Z_t^{u,v}|^2.
    \end{equation*}
    Therefore
    \begin{equation}\label{eq:pairwise-contraction-differential}
        d|Z_t^{u,v}|^2 \le -\lambda |Z_t^{u,v}|^2\,dt + 2\sum_{\ell=1}^m \langle Z_t^{u,v},H_{\ell,t}^{u,v}\rangle\,dB_t^\ell.
    \end{equation}

    Set the new variable
    \begin{equation*}
        Y_t^{u,v}:=e^{\lambda t}|Z_t^{u,v}|^2.
    \end{equation*}
    By It\^o's formula,
    \begin{align*}
        dY_t^{u,v} & = e^{\lambda t} \left[ \lambda |Z_t^{u,v}|^2 + 2\langle Z_t^{u,v},G_t^{u,v}\rangle + \sum_{\ell=1}^m |H_{\ell,t}^{u,v}|^2 \right]dt \\
                   & \quad+ 2e^{\lambda t} \sum_{\ell=1}^m \langle Z_t^{u,v},H_{\ell,t}^{u,v}\rangle\,dB_t^\ell.
    \end{align*}
    By the drift bound \eqref{eq:pairwise-contraction-differential}, the drift term above is nonpositive. Hence
    \begin{equation*}
        dY_t^{u,v} \le 2e^{\lambda t} \sum_{\ell=1}^m \langle Z_t^{u,v},H_{\ell,t}^{u,v}\rangle\,dB_t^\ell.
    \end{equation*}
    Thus \(Y_t^{u,v}\) is a nonnegative local supermartingale.

    To make this completely rigorous, introduce the localization sequence
    \begin{equation*}
        \tau_R := \inf\{t\ge0: |X_t(u)|+|X_t(v)|+|\Theta_t|\ge R\}.
    \end{equation*}
    Since the solution is global and continuous, \(\tau_R\uparrow\infty\) almost surely as \(R\to\infty\). Stopping this inequality at \(\tau_R\), we get
    \begin{equation*}
        Y_{t\wedge\tau_R}^{u,v} \le Y_0^{u,v} + 2\sum_{\ell=1}^m \int_0^{t\wedge\tau_R} e^{\lambda s} \langle Z_s^{u,v},H_{\ell,s}^{u,v}\rangle\,dB_s^\ell.
    \end{equation*}
    The stopped stochastic integral is a true martingale. Therefore, taking
    expectations gives
    \begin{equation*}
        \mathbb E Y_{t\wedge\tau_R}^{u,v} \le Y_0^{u,v}.
    \end{equation*}
    Since \(Y_0^{u,v}=|u-v|^2\), we have
    \begin{equation*}
        \mathbb E\left[ e^{\lambda(t\wedge\tau_R)} |Z_{t\wedge\tau_R}^{u,v}|^2 \right] \le |u-v|^2.
    \end{equation*}
    Letting \(R\to\infty\) and using Fatou's lemma yields
    \begin{equation*}
        \mathbb E\left[e^{\lambda t}|Z_t^{u,v}|^2\right] \le |u-v|^2.
    \end{equation*}
    Therefore
    \begin{equation*}
        \mathbb E|X_t(u)-X_t(v)|^2 \le  e^{-\lambda t}|u-v|^2.
    \end{equation*}
    This proves the pairwise synchronization estimate.

    Now integrate the pairwise synchronization estimate with respect to
    \(\mu_0(du)\mu_0(dv)\). Since \(\mu_0\in\mathcal P_2(\mathbb{R}^d)\),
    \begin{equation*}
        \mathcal D_0 = \iint |u-v|^2\,\mu_0(du)\mu_0(dv)<\infty.
    \end{equation*}
    By Fubini's theorem,
    \begin{align*}
        \mathbb E\mathcal D_t & = \mathbb E \iint |X_t(u)-X_t(v)|^2\,\mu_0(du)\mu_0(dv) \\
                              & = \iint \mathbb E|X_t(u)-X_t(v)|^2\,\mu_0(du)\mu_0(dv)  \\
                              & \le e^{-\lambda t} \iint |u-v|^2\,\mu_0(du)\mu_0(dv).
    \end{align*}
    Hence \(\mathbb E\mathcal D_t\le e^{-\lambda t}\mathcal D_0\), proving
    the first estimate in the theorem.

    Therefore
    \begin{equation*}
        \mathbb E\mathcal V_t = \frac12\mathbb E\mathcal D_t \le \frac12 e^{-\lambda t}\mathcal D_0 = e^{-\lambda t}\mathcal V_0.
    \end{equation*}
    This proves the internal fluctuation decay estimate.

    Finally, since \(\mu_t =\mu_0\circ X_t^{-1}\),
    \begin{equation*}
        \int_{\mathbb{R}^d}|x-\bar X_t|^2\,\mu_t(dx) = \int_{\mathbb{R}^d}|X_t(u)-\bar X_t|^2\,\mu_0(du) = \mathcal V_t.
    \end{equation*}
    For the Dirac measure \(\delta_{\bar X_t}\),
    \begin{equation*}
        W_2^2(\mu_t,\delta_{\bar X_t}) = \int_{\mathbb{R}^d}|x-\bar X_t|^2\,\mu_t(dx).
    \end{equation*}
    Therefore
    \begin{equation*}
        W_2^2(\mu_t,\delta_{\bar X_t})=\mathcal V_t.
    \end{equation*}
    Taking expectations and using the previous estimate gives
    \begin{equation*}
        \mathbb E W_2^2(\mu_t,\delta_{\bar X_t}) \le e^{-\lambda t}W_2^2(\mu_0,\delta_{\bar X_0}).
    \end{equation*}
\end{proof}

\section{Random Metric-Measure Geometry of the Flow}\label{sec:random-mm-geometry}

Let
\begin{equation*}
    S_0:=\operatorname{supp}\mu_0,
\end{equation*}
and
\begin{equation*}
    S_t(\omega):=\operatorname{supp}\mu_t(\omega).
\end{equation*}

For \(x,y\in S_t(\omega)\), define
\begin{equation}\label{eq:transported-metric}
    d_t^\omega(x,y) := \left| T_t(\omega,\cdot)^{-1}(x) - T_t(\omega,\cdot)^{-1}(y)         \right|.
\end{equation}

\begin{theorem}[Random metric-measure geometry of the SDEWI flow]\label{thm:random-mm-geometry}
    Fix \(t\ge0\). Assume that there is an event \(\Omega_t\) of probability
    one such that, for every \(\omega\in\Omega_t\), the map
    \begin{equation*}
        T_t(\omega,\cdot):\mathbb{R}^d\to\mathbb{R}^d
    \end{equation*}
    is a homeomorphism. Then, pathwise for every \(\omega\in\Omega_t\),
    \begin{equation*}
        T_t(\omega): (S_0,|\cdot|,\mu_0) \longrightarrow (S_t(\omega),d_t^\omega,\mu_t(\omega))
    \end{equation*}
    is an isomorphism of metric-measure spaces. More precisely,
    \begin{align}
        S_t(\omega)                             & = T_t(\omega,S_0), \label{eq:support-transport}           \\
        d_t^\omega(T_t(\omega,u),T_t(\omega,v)) & = |u-v|, \qquad u,v\in S_0, \label{eq:isometry-transport} \\
        \mu_t(\omega)                           & = T_t(\omega)_\#\mu_0. \label{eq:measure-transport}
    \end{align}
\end{theorem}

\begin{proof}
    Fix \(t\ge0\) and \(\omega\in\Omega_t\). Write
    \begin{equation*}
        T:=T_t(\omega,\cdot), \qquad \mu:=\mu_0,
        \qquad
        \nu:=\mu_t(\omega)=T_\#\mu,
    \end{equation*}
    and
    \begin{equation*}
        S_0:=\operatorname{supp}\mu,
        \qquad
        S_t:=\operatorname{supp}\nu.
    \end{equation*}

    First, \(S_t=T(S_0)\). If \(x=T(u)\) with \(u\in S_0\), then every open
    set \(U\ni x\) has \(T^{-1}(U)\) open and containing \(u\).
    Hence \(\mu(T^{-1}(U))>0\), so \(\nu(U)>0\), and \(x\in S_t\). Thus
    \(T(S_0)\subset S_t\).
    Conversely, if \(x\notin T(S_0)\), then since \(T\) is a homeomorphism and \(S_0\) is closed, there is an open
    neighborhood \(U\ni x\) with \(U\cap T(S_0)=\varnothing\). Then \(T^{-1}(U)\cap S_0=\varnothing\), so \(\mu(T^{-1}(U))=0\) and
    \(\nu(U)=0\). Hence \(x\notin S_t\), proving
    \eqref{eq:support-transport}.

    The formula \eqref{eq:transported-metric} defines a genuine metric on
    \(S_t\), because \(T:S_0\to S_t\) is bijective and Euclidean distance is
    a metric on \(S_0\). Moreover,
    \begin{equation*}
        d_t^\omega(T(u),T(v)) = |T^{-1}(T(u))-T^{-1}(T(v))| = |u-v|,
    \end{equation*}
    so \(T:(S_0,|\cdot|)\to(S_t,d_t^\omega)\) is an isometry. Since
    \(\nu=T_\#\mu\), \(T\) also preserves the measure. This proves the
    metric-measure isomorphism.
    This completes the proof.
\end{proof}

\subsection{Corollaries}

\begin{corollary}[Diameter and transported balls]\label{cor:diameter-ball-transport}
    Under the assumptions of Theorem~\ref{thm:random-mm-geometry},
    \begin{equation}\label{eq:diameter-transport}
        \operatorname{diam}_{d_t^\omega}S_t(\omega) = \operatorname{diam}S_0,
    \end{equation}
    and for every \(u\in S_0\) and \(r>0\),
    \begin{equation}\label{eq:ball-transport}
        B_{d_t^\omega}(T_t(\omega,u),r) =T_t(\omega,B(u,r)\cap S_0).
    \end{equation}
\end{corollary}

\begin{proof}
    Write \(T:=T_t(\omega,\cdot)\). By Theorem~\ref{thm:random-mm-geometry}, \(T:(S_0,|\cdot|)\to
    (S_t(\omega),d_t^\omega)\) is an isometry.

    For \(u\in S_0\) and \(r>0\), a point \(x\in S_t(\omega)\) belongs to
    \(B_{d_t^\omega}(T(u),r)\) if and only if \(x=T(v)\) for some
    \(v\in S_0\) with \(|v-u|<r\). Hence
    \begin{equation*}
        B_{d_t^\omega}(T(u),r) = T(B(u,r)\cap S_0).
    \end{equation*}
    This proves \eqref{eq:ball-transport}. The diameter identity
    \eqref{eq:diameter-transport} follows by taking suprema over pairs.
\end{proof}

\begin{corollary}[Measure of transported balls]\label{cor:ball-measure-transport}
    Under the assumptions of Theorem~\ref{thm:random-mm-geometry}, for every
    \(u\in S_0\) and \(r>0\),
    \begin{equation}
        \mu_t(\omega)\big(B_{d_t^\omega}(T_t(\omega,u),r)\big)
        =
        \mu_0(B(u,r)\cap S_0).
        \label{eq:ball-measure-transport}
    \end{equation}
\end{corollary}

\begin{proof}
    By \eqref{eq:ball-transport} and
    \(\mu_t(\omega)=T_t(\omega)_\#\mu_0\),
    \begin{equation*}
        \mu_t(\omega)\big(B_{d_t^\omega}(T_t(\omega,u),r)\big) =
        \mu_t(\omega)\big(T_t(\omega,B(u,r)\cap S_0)\big) = \mu_0(B(u,r)\cap S_0).
    \end{equation*}
\end{proof}

\begin{corollary}[Metric-measure inequalities in the transported metric]\label{cor:transported-metric-inequalities}
    Under the assumptions of Theorem~\ref{thm:random-mm-geometry}, every
    purely metric-measure property is preserved exactly under the isomorphism
    \begin{equation*}
        (S_0,|\cdot|,\mu_0) \simeq (S_t(\omega),d_t^\omega,\mu_t(\omega)).
    \end{equation*}
    In particular, concentration, doubling, isoperimetric, and Cheeger
    properties are preserved in the transported metric. Poincar\'e,
    logarithmic Sobolev, and Talagrand inequalities are also preserved with
    the same constants.

    If \(\mu_0\) satisfies \(T_p(C_T)\), then \(\mu_t(\omega)\) satisfies
    \(T_p(C_T)\) in the random metric \(d_t^\omega\). If, in addition,
    \(T_t(\omega,\cdot)\) is \(L\)-Lipschitz on \(S_0\), then
    \(\mu_t(\omega)\) satisfies the Euclidean transport inequality
    \(T_p(C_TL^2)\).
\end{corollary}

\begin{proof}
    Write \(T:=T_t(\omega,\cdot)\), \(\mu:=\mu_0\), and
    \(\nu:=\mu_t(\omega)\). By Theorem~\ref{thm:random-mm-geometry},
    \(T:(S_0,|\cdot|,\mu)\to(S_t,d_t^\omega,\nu)\) is a metric-measure
    isomorphism.

    For example, \(h:S_t\to\mathbb{R}\) is \(1\)-Lipschitz with respect to
    \(d_t^\omega\) if and only if \(h\circ T:S_0\to\mathbb{R}\) is
    \(1\)-Lipschitz in the Euclidean metric. Hence concentration inequalities
    for Euclidean \(1\)-Lipschitz functions under \(\mu_0\) transfer exactly
    to \(d_t^\omega\)-Lipschitz functions under \(\mu_t(\omega)\). The same
    invariance argument applies to the other purely metric-measure
    properties.

    For transport inequalities, let \(\alpha,\beta\) be probability measures
    on \(S_t\), and set
    \begin{equation*}
        \alpha_0:=(T^{-1})_\#\alpha,
        \qquad
        \beta_0:=(T^{-1})_\#\beta.
    \end{equation*}
    Couplings are transported bijectively by \(T\), and the transportation
    costs are identical:
    \begin{equation*}
        W_{p,d_t^\omega}(\alpha,\beta)
        =
        W_p(\alpha_0,\beta_0).
    \end{equation*}
    Relative entropy is also invariant under pushforward:
    \begin{equation*}
        \operatorname{Ent}(\alpha\mid\nu)
        =
        \operatorname{Ent}(\alpha_0\mid\mu).
    \end{equation*}
    Therefore any \(T_p(C_T)\) inequality for \(\mu\) transfers to \(\nu\) in
    the metric \(d_t^\omega\). If \(T\) is \(L\)-Lipschitz on \(S_0\), then
    \begin{equation*}
        |x-y|\le L\,d_t^\omega(x,y),
        \qquad x,y\in S_t,
    \end{equation*}
    and the Euclidean \(T_p(C_TL^2)\) estimate follows.
\end{proof}

\begin{corollary}[Weighted functional inequalities]\label{cor:weighted-functional-inequalities}
    Assume the hypotheses of Theorem~\ref{thm:random-mm-geometry}. Suppose,
    in addition, that \(T_t(\omega,\cdot)\) is a \(C^1\)-diffeomorphism and
    define
    \begin{equation*}
        J_t(u,\omega):=D_uT_t(\omega,u).
    \end{equation*}
    For \(x=T_t(\omega,u)\), define the co-metric matrix
    \begin{equation}\label{eq:random-cometric}
        G_t^\omega(x) := J_t(u,\omega)J_t(u,\omega)^\top.
    \end{equation}
    If \(\mu_0\) satisfies \(\mathrm{LSI}(C_0)\), then \(\mu_t(\omega)\)
    satisfies the weighted logarithmic Sobolev inequality
    \begin{equation}\label{eq:weighted-lsi}
        \operatorname{Ent}_{\mu_t(\omega)}(h^2)
        \le
        C_0
        \int
        \left\langle G_t^\omega(x)\nabla h(x), \nabla h(x) \right\rangle
        \,\mu_t(\omega)(dx).
    \end{equation}
    If \(\mu_0\) satisfies \(\mathrm{PI}(C_P)\), then
    \begin{equation}\label{eq:weighted-pi}
        \operatorname{Var}_{\mu_t(\omega)}(h)
        \le C_P \int \left\langle G_t^\omega(x)\nabla h(x), \nabla h(x) \right\rangle
        \,\mu_t(\omega)(dx).
    \end{equation}
    Finally, if
    \begin{equation}\label{eq:flow-lip-constant}
        \mathcal K_t(\omega;S_0) := \sup_{u\in S_0} \|J_t(u,\omega)\|_{\mathrm{op}} <\infty,
    \end{equation}
    then the ordinary Euclidean inequalities follow:
    \begin{equation*}
        \mathrm{LSI}(C_0)
        \Longrightarrow
        \mu_t(\omega)\in
        \mathrm{LSI}\big(C_0\mathcal K_t(\omega;S_0)^2\big),
    \end{equation*}
    and
    \begin{equation*}
        \mathrm{PI}(C_P)
        \Longrightarrow
        \mu_t(\omega)\in
        \mathrm{PI}\big(C_P\mathcal K_t(\omega;S_0)^2\big).
    \end{equation*}
\end{corollary}

\begin{proof}
    Write \(T:=T_t(\omega,\cdot)\), \(\mu:=\mu_0\), and
    \(\nu:=\mu_t(\omega)\). Put \(J(u):=D_uT(u)\). If \(x=T(u)\), let
    \begin{equation*}
        G(x)=J(u)J(u)^\top.
    \end{equation*}
    Let \(h\in C_c^1(\mathbb{R}^d)\) and set \(f=h\circ T\). Entropy is
    invariant under the pushforward, so
    \begin{equation*}
        \operatorname{Ent}_{\nu}(h^2)
        =
        \operatorname{Ent}_{\mu}(f^2).
    \end{equation*}
    If \(\mu\) satisfies \(\mathrm{LSI}(C_0)\), then
    \begin{equation*}
        \operatorname{Ent}_{\nu}(h^2)
        \le
        C_0\int |\nabla f(u)|^2\,\mu(du).
    \end{equation*}
    By the chain rule,
    \begin{equation*}
        \nabla f(u)=J(u)^\top\nabla h(T(u)),
    \end{equation*}
    so
    \begin{equation*}
        |\nabla f(u)|^2
        =
        \left\langle
        J(u)J(u)^\top\nabla h(T(u)),
        \nabla h(T(u))
        \right\rangle.
    \end{equation*}
    Pushing forward by \(T\) proves \eqref{eq:weighted-lsi}. The proof of
    \eqref{eq:weighted-pi} is identical, with entropy replaced by variance.

    If \(L=\sup_{u\in S_0}\|J(u)\|_{\mathrm{op}}\), then
    \begin{equation*}
        \left\langle G(x)\nabla h(x),\nabla h(x)\right\rangle
        =
        |J(u)^\top\nabla h(x)|^2
        \le
        L^2|\nabla h(x)|^2,
    \end{equation*}
    giving the Euclidean LSI and Poincar\'e constants.
\end{proof}

\begin{remark}
    The ordinary Euclidean geometry of \(\mu_t(\omega)\) may stretch,
    contract, or become anisotropic, but the intrinsic random metric geometry
    is exactly the initial geometry transported by the stochastic flow:
    \begin{equation*}
        \text{common noise transports the initial metric-measure space; it does not erase it.}
    \end{equation*}
\end{remark}
