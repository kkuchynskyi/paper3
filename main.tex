\begin{abstract}
	We study the measure-valued solution to the stochastic differential equations with interaction.
	Under regularity assumptions,  we obtain moment estimates for
	the maximal Jacobian norm on compact sets, which imply the functional inequalities.
	We also establish an averaged Jacobian-energy bound under one-sided derivative assumptions.
	Finally, for a kernel-type interacting model satisfying a pairwise dissipativity condition,
	we prove exponential decay of the expected pairwise variance, yielding convergence to the center of mass.
\end{abstract}

\section{Introduction}\label{sec:introduction}

% The definition of SDEWI
We consider stochastic differential equations with interaction that were proposed by Andrey A. Dorogovtsev in \cite{dorogovtsev2003},
%  Main SDEWIN equation
\begin{equation}\label{eq:main-sdewi}
	\begin{cases}
		dX_t(u) & = a\big(X_t(u),\mu_t\big)\,dt + \displaystyle\sum_{\ell=1}^{r} \sigma_\ell\bigl(X_t(u))\,dB_t^\ell, \\[1mm]
		X_0(u)  & = u,                                                                                                \\[1mm]
		\mu_t   & = \mu_0\circ X_t^{-1}.
	\end{cases}
\end{equation}

% The details of the SDEWI
Here \( \mu_0\in\mathcal P_2(\mathbb R^d).\) is a deterministic initial probability measure, where \(\cP_2(\R^d)\) denotes
the space of probability measures on \(\R^d\) with finite second moment.
The drift is \(a:\R^d\times\cP_2(\R^d)\to\R^d\), and the diffusion
coefficients \(\sigma_\ell:\R^d\to\R^d\), \(\ell=1,\ldots,r\), are state-dependent and non-constant.
Let\( (\Omega,(\mathcal F_t)_{t\ge0},\mathbb P) \) be a filtered probability space satisfying the usual
conditions and carrying a \(r\)-dimensional Brownian motion \(B=(B^1,\dots,B^r) \).

% Assumption on the coefficients
\begin{assumption}[Well-posedness]\label{ass:wellposedness}
	There exists \(L<\infty\) such that, for all
	\(x,y\in\R^d\) and
	\(\mu,\nu\in\cP_2(\R^d)\),
	\begin{equation*}
		|a(x,\mu)-a(y,\nu)| + \left( \sum_{\ell=1}^{r} |\sigma_\ell(x)-\sigma_\ell(y)|^2 \right)^{1/2}  \le L\bigl(|x-y|+W_2(\mu,\nu)\bigr)
	\end{equation*}
\end{assumption}

% The remarks on the solution
Under \Cref{ass:wellposedness}, the SDEWI~\eqref{eq:main-sdewi} admits a unique global solution; see
\cite[Theorem~2.3.1]{dorogovtsev2023}.

% a random transport and a jacobian
Unlike a classical stochastic differential equation, the drift depends on the push-forward of the entire system through the random measure
\(\mu_t \).
This dependence represents the interaction between particles. For each realization \(\omega \in \Omega\),
the random map
\begin{equation*}
	T_t(\omega,\cdot):u\mapsto X_t(u,\omega)
\end{equation*}
acts on the initial measure by pushforward,
\begin{equation*}
	\mu_t(\omega)=\mu_0\circ T_t(\omega)^{-1},
\end{equation*}
so the evolution is can be viewed as a random transport of measures by a stochastic flow \eqref{eq:main-sdewi}. Whenever the flow is
differentiable with respect to the initial label, we write
\begin{equation*}
	J_t(u):=\partial_u T_t(u),
\end{equation*}
and consider the three quantities described below. Throughout the paper, \(|\cdot|\) denotes the Euclidean norm and
inner product on finite-dimensional vector spaces. For matrices \(M,N\) of the same size, we write
\begin{equation*}
	\|M\| := \bigl(\operatorname{Tr}(M^{\mathsf T}M)\bigr)^{1/2}, \qquad \langle M,N\rangle := \operatorname{Tr}(M^{\mathsf T}N).
\end{equation*}
Thus \(\|\cdot\|\) is always the Hilbert--Schmidt norm when its argument is a matrix.

% Section 2 motivation 
In \Cref{sec:jacobian-bound}, we study \(\mathcal K_t(K)\)  on the compact set \(K\),
Firstly, we consider the maximal Jacobian norm \(\mathcal K_t(K)\)
\begin{equation*}
	\mathcal K_t(K):=\sup_{u\in K}\|J_t(u)\|.
\end{equation*}
We prove moment estimates for this random constant on compact sets and
show that it governs the pathwise transfer of logarithmic Sobolev, concentration, and transport inequalities from \(\mu_0\) to
\(\mu_t\).
% Logarithmic Sobolev inequality
\begin{definition}[Logarithmic Sobolev inequality]
	Denoting by \(\cC_c^1(\R^d)\) the set of continuously differentiable compactly supported functions from \(\R^d\) to \(\R\), for
	\(g\ge 0\) with \(\int g\,d\mu=1\) we define the entropy
	\begin{equation*}
		\Ent_\mu(g):=\int g\log g\,d\mu.
	\end{equation*}
	A probability measure \(\mu\) on \(\R^d\) is said to satisfy a logarithmic Sobolev inequality with constant \(C>0\) if for every
	\(h\in\cC_c^1(\R^d)\) with \(\int h^2\,d\mu=1\) one has
	\begin{equation*}
		\Ent_\mu(h^2)\le C\int_{\R^d}|\partial h|^2\,d\mu.
	\end{equation*}
\end{definition}

% Talagrand transport inequality
\begin{definition}[Talagrand transport inequality]
	Let \(p\ge1\). A probability measure \(\mu\) on \(\R^d\) is said to satisfy the Talagrand transport inequality \(T_p(C_T)\) with constant
	\(C_T>0\) if for every probability measure \(\nu\) on \(\R^d\) with finite \(p\)-th moment,
	\begin{equation*}
		W_p(\nu,\mu)^2 \le 2C_T\,\Ent(\nu\mid \mu),
	\end{equation*}
	where
	\begin{equation*}
		\Ent(\nu\mid\mu)=
		\begin{cases}
			\displaystyle \int_{\R^d}\log\Big(\frac{d\nu}{d\mu}\Big)\,d\nu, & \nu\ll \mu,       \\[3mm]
			+\infty,                                                        & \text{otherwise}.
		\end{cases}
	\end{equation*}
\end{definition}

% Section 3 motivation
In \Cref{sec:averaged-jacobian}, we replace the supremum over labels by the averaged Jacobian energy \(\mathcal E_t\). T
% The Jacobian energy 
The Jacobian energy \(\mathcal E_t\)
\begin{equation*}
	\mathcal E_t := \int_{\R^d}\|J_t(u)\|^2\,\mu_0(du),
\end{equation*}
over the initial mass \( \mu_0 \). Averaged Jacobian energy can be estimated under weaker one-sided
derivative conditions.

% Section 4 summary
Finally, in \Cref{sec:internal-fluctuations}, we exploit the specific kernel-type interacting model,
we impose a pairwise dissipativity condition and prove exponential decay of \(\E\mathcal D_t\).
This decay means that the transported labels approach one another in mean square and that the random measure synchronizes around its random
center of mass.

% The pairwise energy
The pairwise energy \(\mathcal D_t\)
\begin{equation*}
	\mathcal D_t :=	\iint_{\R^d\times\R^d} |T_t(u)-T_t(v)|^2\,\mu_0(du)\mu_0(dv).
\end{equation*}
measures deformation at finite scale by averaging the squared distance between two transported labels.
Pairwise energy can be controlled directly through dissipativity of the dynamics, without obtaining a uniform Jacobian estimate

\section{Pathwise Jacobian Bounds and Functional Inequality Transfer}\label{sec:jacobian-bound}

% Roadmap to the section
Firstly, we establish \(\mathcal K_t(K)\), then use it to transfer functional inequalities from the initial measure \( \mu_0 \) to the
random pushforward measure \( \mu_t \). The technical proof of the compact-set Jacobian estimate is postponed to the final subsection.

% The assumptions for this section on the Jacobian
\begin{assumption}[Spatial differentiability]\label{ass:flow-differentiability}
	For every \(\mu\in\cP_2(\R^d)\), the map \(x\mapsto a(x,\mu)\) belongs to
	\(\cC^1(\R^d;\R^d)\), and
	\begin{equation*}
		\sigma_\ell\in \cC^1(\R^d;\R^d), \qquad \ell=1,\dots,r.
	\end{equation*}
\end{assumption}

By \Cref{ass:wellposedness,ass:flow-differentiability}, for all
\(x,z\in\R^d\) and \(\mu\in\cP_2(\R^d)\),
\begin{equation}\label{eq:basic-derivative-bounds}
	|\partial_x a(x,\mu)z| \le L_x|z|, \qquad \sum_{\ell=1}^{r} |\partial_x\sigma_\ell(x)z|^2 \le L_\sigma^2|z|^2.
\end{equation}
for all \(x,z\in\R^d\) and \(\mu\in\cP_2(\R^d)\). Here one may take \(L_x=L_\sigma=L\).

\subsection{The Jacobian flow}\label{subsec:jacobian-flow-distortion-constants}

% Roadmap to the subsection
We use the Jacobian SDE from \cite{dorogovtsev_weiss2023}. For \eqref{eq:main-sdewi}, it has the following form.

% Jacobian SDE
\begin{theorem}[Jacobian equation] \label{thm:jacobian-sde}
	Assume \Cref{ass:wellposedness,ass:flow-differentiability}. Then the solution admits a
	modification such that, for almost every \(\omega\) and every \(t\ge0\)
	its Jacobian \( J_t(u) \) satisfies
	\begin{equation*}
		\begin{cases}
			dJ_t(u) = A_t(u)J_t(u)\,dt + \sum_{\ell=1}^{r} C_{\ell,t}(u)J_t(u)\,dB_t^\ell, \\
			J_0(u)=I,                                                                      \\
		\end{cases}
	\end{equation*}
	where
	\begin{equation*}
		A_t(u):=\partial_x a\bigl(X_t(u),\mu_t\bigr), \qquad C_{\ell,t}(u):=\partial_x\sigma_\ell\bigl(X_t(u)\bigr).
	\end{equation*}
\end{theorem}

% The definition of Random Jacobian constant
\begin{definition}[Random Jacobian constant]\label{def:jacobian-constant}
	For a nonempty set \(K\subset\R^d\), define
	\begin{equation*}
		\mathcal K_t(K):=\sup_{u\in K}\|J_t(u)\|.
	\end{equation*}
\end{definition}

% Remark
If \(K\) is compact, continuity of \(u\mapsto J_t(u)\) implies \(\mathcal K_t(K)<\infty\).
This compact-set quantity is used because a corresponding supremum over all of \(\R^d\) need not be finite.
The counterexamples of Mohammed--Scheutzow show that even for globally Lipschitz coefficients one may fail to have a finite global constant
\( \sup_{\substack{u\in\R^d\\ z\in\R^d,\ |z|=1}} |J_t(u,\omega)z| \) for a fixed \((t,\omega)\).

% Intro for the theorem
% Controlling the compact-set maximal Jacobian norm \(\mathcal K_t(K)\) requires estimates for the spatial increments \(J_t(u)-J_t(v)\).

\begin{theorem}[Moments of the compact-set maximal Jacobian norm]\label{thm:kt-moment}
	Assume \Cref{ass:wellposedness,ass:flow-differentiability}. Suppose in addition that there exists a constant
	\(H<\infty\) such that, for all \(x,y,z\in\R^d\) and \(\mu\in\cP_2(\R^d)\),
	\begin{equation}\label{eq:spatial-lipschitz-derivatives}
		\big| \big(\partial_x a(x,\mu)-\partial_x a(y,\mu)\big)z \big| + \left( \sum_{\ell=1}^{r}
		\big| \big(\partial_x\sigma_\ell(x)-\partial_x\sigma_\ell(y)\big)z \big|^2 \right)^{1/2}
		\le H|x-y|\,|z|.
	\end{equation}
	Then, for every nonempty compact \(K\subset\R^d\) and every
	\(p\ge1\), there exist constants \(C_{p,K},c_p<\infty\) such that
	\begin{equation}\label{eq:kt-exp-growth}
		\E\big[\mathcal K_t(K)^p\big] \le C_{p,K}e^{c_pt}, \qquad t\ge0.
	\end{equation}
\end{theorem}

The proof is given in \Cref{subsec:kt-moment-proof}.

\subsection{Transfer of functional inequalities}\label{subsec:functional-inequality-transfer}

Throughout this subsection, assume that \(\supp(\mu_0)\) is compact and set
\begin{equation*}
	K_0:=\operatorname{co}\bigl(\supp(\mu_0)\bigr), \qquad \kappa_t(\omega):=\mathcal K_t(K_0).
\end{equation*}
Since \(K_0\) is compact and convex, for every fixed \(t\ge0\) and almost every \(\omega\),
\begin{equation}\label{eq:flow-lipschitz-kappa}
	\Lip\bigl(T_t(\omega,\cdot)|_{K_0}\bigr) \le \kappa_t(\omega).
\end{equation}
Indeed, for \(u,v\in K_0\),
\begin{align*}
	|T_t(\omega,u)-T_t(\omega,v)| & \le \int_0^1\left| J_t\bigl(v+s(u-v),\omega\bigr)(u-v) \right|\,ds \\
	                              & \le \kappa_t(\omega)|u-v|.
\end{align*}
because \(v+s(u-v)\in K_0\) for every \(s\in[0,1]\).

\begin{corollary}[Pathwise logarithmic Sobolev inequality]\label{cor:pathwise-lsi}
	Suppose that \(\mu_0\) satisfies \(\mathrm{LSI}(C_0)\).
	Then, for every fixed \(t\ge0\) and almost every \(\omega\),
	\begin{equation*}
		\mu_t(\omega)=\mu_0\circ T_t(\omega,\cdot)^{-1}
	\end{equation*}
	satisfies
	\begin{equation*}
		\mathrm{LSI}\bigl(C_0\kappa_t(\omega)^2\bigr).
	\end{equation*}
\end{corollary}

\begin{proof}
	Fix \(t\ge0\) and \(\omega\) such that
	\(u\mapsto T_t(\omega,u)\) is of class \(\cC^1\), and write
	\begin{equation*}
		T:=T_t(\omega,\cdot), \qquad J(u):=J_t(u,\omega), \qquad \kappa:=\kappa_t(\omega).
	\end{equation*}

	Let \(g\in\cC_c^1(\R^d)\) satisfy
	\begin{equation*}
		\int_{\R^d}g^2\,d\mu_t(\omega)=1.
	\end{equation*}
	Choose \(\chi\in\cC_c^1(\R^d)\) such that
	\(\chi=1\) on a neighborhood of \(K_0\), and set
	\begin{equation*}
		h(u):=\chi(u)g(T(u)).
	\end{equation*}
	Since \(\mu_0\) is supported on \(K_0\),
	\begin{equation*}
		h(u)=g(T(u)), \qquad \partial h(u)=J(u)^{\mathsf T}\partial g(T(u)), \qquad u\in K_0.
	\end{equation*}
	Consequently,
	\begin{align*}
		\Ent_{\mu_t(\omega)}(g^2) & = \Ent_{\mu_0}(h^2)                                                         \\
		                          & \le C_0\int_{K_0}\left|J(u)^{\mathsf T}\partial g(T(u))\right|^2\,\mu_0(du) \\
		                          & \le C_0\kappa^2\int_{K_0}|\partial g(T(u))|^2\,\mu_0(du)                    \\
		                          & = C_0\kappa^2\int_{\R^d}|\partial g(x)|^2\,\mu_t(\omega,dx).
	\end{align*}
\end{proof}

\begin{proposition}[Pathwise concentration transfer]\label{prop:concentration-transfer}
	Suppose that \(\mu_0\) satisfies
	\begin{equation*}
		\mu_0\bigl(f-\mu_0(f)\ge r\bigr) \le \Psi(r), \qquad r\ge0,
	\end{equation*}
	for every \(1\)-Lipschitz function
	\(f:\R^d\to\R\), where
	\(\Psi:[0,\infty)\to[0,1]\) is nonincreasing.

	Then, for every fixed \(t\ge0\), almost every \(\omega\), every
	\(1\)-Lipschitz function \(g:\R^d\to\R\), and every \(r>0\),
	\begin{equation*}
		\mu_t(\omega)\bigl(g-\mu_t(\omega)(g)\ge r\bigr) \le \Psi\left(\frac{r}{\kappa_t(\omega)}\right),
	\end{equation*}
	whenever \(\kappa_t(\omega)>0\).

	If \(\kappa_t(\omega)=0\), then \(\mu_t(\omega)\) is a Dirac
	mass and the probability on the left-hand side is zero.
\end{proposition}

\begin{proof}
	Fix \(t\ge0\) and \(\omega\) for which
	\eqref{eq:flow-lipschitz-kappa} holds, and write
	\begin{equation*}
		T:=T_t(\omega,\cdot), \qquad \kappa:=\kappa_t(\omega).
	\end{equation*}

	If \(\kappa=0\), then
	\eqref{eq:flow-lipschitz-kappa} shows that \(T\) is constant
	on \(K_0\). Since \(\mu_0\) is supported on \(K_0\),
	\(\mu_t(\omega)\) is a Dirac mass.

	Suppose that \(\kappa>0\), let \(g:\R^d\to\R\) be
	\(1\)-Lipschitz, and define on \(K_0\)
	\begin{equation*}
		h(u):=\frac{g(T(u))}{\kappa}.
	\end{equation*}
	By \eqref{eq:flow-lipschitz-kappa}, \(h\) is \(1\)-Lipschitz
	on \(K_0\). By the McShane extension theorem, there exists a
	\(1\)-Lipschitz function
	\(\widetilde h:\R^d\to\R\) such that
	\(\widetilde h=h\) on \(K_0\).

	Since \(\mu_0(K_0)=1\),
	\begin{equation*}
		\mu_0(\widetilde h)=\mu_0(h)=\frac{1}{\kappa}\mu_t(\omega)(g).
	\end{equation*}
	Therefore,
	\begin{align*}
		\mu_t(\omega)\bigl(g-\mu_t(\omega)(g)\ge r\bigr) & = \mu_0\left(\widetilde h-\mu_0(\widetilde h)\ge\frac{r}{\kappa}\right) \\
		                                                 & \le \Psi\left(\frac{r}{\kappa}\right).
	\end{align*}
\end{proof}

\begin{remark}[Concentration as a consequence of LSI]
	If \(\mu_0\) satisfies a logarithmic Sobolev inequality, then
	it satisfies a Gaussian concentration inequality. Therefore,
	\Cref{cor:pathwise-lsi} yields a pathwise Gaussian concentration
	inequality for \(\mu_t(\omega)\), with random concentration
	constant controlled by
	\begin{equation*}
		\kappa_t(\omega)^2=\mathcal K_t(K_0)^2.
	\end{equation*}
\end{remark}

We next record the analogous stability property for Talagrand transport inequalities.

\begin{proposition}[Transport inequality under a localized Lipschitz pushforward]\label{prop:talagrand-pushforward}
	Let \(\mu\) be a probability measure satisfying \(T_p(C_T)\)
	for some \(p\ge1\), and suppose that
	\(\supp(\mu)\subset K\) for some compact set
	\(K\subset\R^d\).

	Let \(T:\R^d\to\R^d\) be a Borel map whose restriction to \(K\)
	is Lipschitz, and set
	\begin{equation*}
		L_K(T):=\sup_{\substack{u,v\in K\\u\neq v}}\frac{|T(u)-T(v)|}{|u-v|}.
	\end{equation*}
	Then the pushforward measure
	\(\eta:=\mu\circ T^{-1}\) satisfies
	\begin{equation*}
		T_p\bigl(C_T L_K(T)^2\bigr).
	\end{equation*}
\end{proposition}

\begin{proof}
	Let \(\lambda\) be a probability measure on \(\R^d\).
	If \(\lambda\not\ll\eta\), then
	\(\Ent(\lambda\mid\eta)=+\infty\), and there is nothing to prove.

	Suppose that \(\lambda\ll\eta\), write
	\(\varphi:=d\lambda/d\eta\), and define
	\begin{equation*}
		\widetilde\lambda(du):=\varphi(T(u))\,\mu(du).
	\end{equation*}
	Then
	\begin{equation*}
		\widetilde\lambda\circ T^{-1}=\lambda, \qquad \Ent(\widetilde\lambda\mid\mu)=\Ent(\lambda\mid\eta).
	\end{equation*}

	Moreover,
	\(\widetilde\lambda(K)=\mu(K)=1\). Hence every coupling
	\(\pi\) of \(\widetilde\lambda\) and \(\mu\) is concentrated on
	\(K\times K\), and therefore
	\begin{align*}
		\int|T(u)-T(v)|^p\,\pi(du,dv) & \le L_K(T)^p\int|u-v|^p\,\pi(du,dv).
	\end{align*}
	Taking the infimum over \(\pi\) gives
	\begin{equation*}
		W_p(\lambda,\eta) \le L_K(T)W_p(\widetilde\lambda,\mu).
	\end{equation*}
	Consequently,
	\begin{align*}
		W_p(\lambda,\eta)^2 & \le L_K(T)^2W_p(\widetilde\lambda,\mu)^2       \\
		                    & \le 2C_TL_K(T)^2\Ent(\widetilde\lambda\mid\mu) \\
		                    & = 2C_TL_K(T)^2\Ent(\lambda\mid\eta).
	\end{align*}
\end{proof}

\subsection{Proof of the compact-set Jacobian estimate}\label{subsec:kt-moment-proof}

The proof of \Cref{thm:kt-moment} relies on two elementary estimates for linear stochastic equations.

\begin{lemma}\label{lem:linear-sde-moment}
	Let \(Y_t\) solve
	\begin{equation*}
		\begin{cases}
			dY_t = A_tY_t\,dt + \sum_{\ell=1}^{r} C_{\ell,t}Y_t\,dB_t^\ell, \\
			Y_0=y.                                                          \\
		\end{cases}
	\end{equation*}

	Assume that, for every \(z\in\R^d\),
	\begin{equation*}
		\langle z,A_tz\rangle \le L_x|z|^2, \qquad \sum_{\ell=1}^{r}|C_{\ell,t}z|^2 \le L_\sigma^2|z|^2.
	\end{equation*}

	Then, for every \(q\ge1\),
	\begin{equation*}
		\E|Y_t|^{2q} \le |y|^{2q} \exp\left\{ q\bigl(2L_x+(2q-1)L_\sigma^2\bigr)t \right\}.
	\end{equation*}

	Consequently,
	\begin{equation*}
		\E\|J_t(u)\|^{2q} \le d^q \exp\left\{ q\bigl(2L_x+(2q-1)L_\sigma^2\bigr)t \right\}.
	\end{equation*}
\end{lemma}

\begin{proof}
	Applying It\^o's formula to \(|Y_t|^{2q}\), after a standard
	localization, gives
	\begin{align*}
		d|Y_t|^{2q} & = 2q|Y_t|^{2q-2}\langle Y_t,A_tY_t\rangle\,dt  + q|Y_t|^{2q-2} \sum_{\ell=1}^{r}|C_{\ell,t}Y_t|^2\,dt \\
		            & \quad +2q(q-1)|Y_t|^{2q-4} \sum_{\ell=1}^{r} \langle Y_t,C_{\ell,t}Y_t\rangle^2\,dt +dM_t,
	\end{align*}
	where \(M_t\) is a local martingale. Since
	\begin{equation*}
		\langle Y_t,C_{\ell,t}Y_t\rangle^2 \le |Y_t|^2|C_{\ell,t}Y_t|^2,
	\end{equation*}
	the assumptions imply
	\begin{equation*}
		d|Y_t|^{2q} \le q\bigl(2L_x+(2q-1)L_\sigma^2\bigr) |Y_t|^{2q}\,dt+dM_t.
	\end{equation*}
	Taking expectations in the localized equation and applying Gronwall's
	lemma yields
	\begin{equation*}
		\E|Y_t|^{2q} \le |y|^{2q} \exp\left\{ q\bigl(2L_x+(2q-1)L_\sigma^2\bigr)t \right\}.
	\end{equation*}

	For the matrix estimate, apply the vector estimate to
	\(J_t(u)e_j\), \(j=1,\dots,d\), and use
	\begin{equation*}
		\|J_t(u)\|^{2q} = \left(\sum_{j=1}^{d}|J_t(u)e_j|^2\right)^q \le d^{q-1}\sum_{j=1}^{d}|J_t(u)e_j|^{2q}.
	\end{equation*}
	This gives
	\begin{equation*}
		\E\|J_t(u)\|^{2q} \le d^q \exp\left\{ q\bigl(2L_x+(2q-1)L_\sigma^2\bigr)t\right\}.
	\end{equation*}
\end{proof}

% The preceding lemma applies directly to \(X_t(u)-X_t(v)\), because the difference of two trajectories driven by the same Brownian motion
% satisfies a homogeneous linear equation with random coefficients. The difference \(J_t(u)-J_t(v)\) satisfies an inhomogeneous linear equation:
% additional terms arise from evaluating the derivative coefficients along the two different trajectories. We therefore record the corresponding
% moment estimate.

\begin{lemma}\label{lem:inhomogeneous-linear-moment}
	Let \(m\ge2\) be an even integer, and suppose that the
	matrix-valued process \(D_t\) satisfies
	\begin{equation*}
		dD_t = A_tD_t\,dt + \sum_{\ell=1}^{r}C_{\ell,t}D_t\,dB_t^\ell + F_t\,dt + \sum_{\ell=1}^{r}G_{\ell,t}\,dB_t^\ell.
	\end{equation*}
	Assume that, for every \(M\in\R^{d\times d}\),
	\begin{equation*}
		\langle M,A_tM\rangle \le L_x\|M\|^2, \qquad \sum_{\ell=1}^{r}\|C_{\ell,t}M\|^2 \le L_\sigma^2\|M\|^2.
	\end{equation*}
	Then there exist constants \(C_m,c_m<\infty\), depending only on
	\(m,L_x,L_\sigma\), such that
	\begin{align*}
		\E\|D_t\|^m & \le e^{c_mt}\E\|D_0\|^m                                                                                                  \\
		            & \quad +C_m\int_0^t e^{c_m(t-s)} \left[ \E\|F_s\|^m + \E\left( \sum_{\ell=1}^{r}\|G_{\ell,s}\|^2 \right)^{m/2} \right]ds.
	\end{align*}
\end{lemma}

% Although the drift depends on the random measure \( \mu_t \), differentiation with respect to the initial label acts only on the spatial argument
% \(X_t(u) \). Because of that fact we can evaluate \(\|J_t(u)\|\) using Kunita's theorem for the stochastic flows.

% Under \Cref{ass:wellposedness,ass:flow-differentiability}, standard results on stochastic flows (see, e.g., \cite[Theorem~1.4.1]{Kunita1986})
% yield a modification such that for a.s.\ \(\omega\) and each \(t\ge0\) the map \(u\mapsto T_t(\omega,u)\) is \(\cC^1\).

\begin{proof}[Proof of \Cref{thm:kt-moment}]

	% fix initial conditions
	Fix a compact set \(K\subset\R^d\), \(t\ge0\), and \(p\ge1\). Choose an even integer \(m>\max\{2p,2d\}\).
	% It is enough to prove
	% \eqref{eq:kt-exp-growth} with \(p\) replaced by \(m\), because then H\"older's inequality gives
	% \begin{equation*}
	% 	\E\,\mathcal K_t(K)^p \le \big(\E\,\mathcal K_t(K)^m\big)^{p/m}.
	% \end{equation*}

	% define the difference \Delta X_t
	Fix \(u,v\in\R^d\) and set
	\begin{equation*}
		\Delta X_t:=X_t(u)-X_t(v).
	\end{equation*}

	% explicit d\Delta X_t
	Since the same random measure \(\mu_t\) appears in both equations, subtraction yields
	\begin{align*}
		d\Delta X_t & = \bigl( a(X_t(u),\mu_t)-a(X_t(v),\mu_t) \bigr)\,dt                                        \\
		            & \quad + \sum_{\ell=1}^{r} \bigl(\sigma_\ell(X_t(u))-\sigma_\ell(X_t(v)) \bigr)\,dB_t^\ell.
	\end{align*}

	% the average coefficients \widetilde A_t(u,v) and \widetilde C_{\ell,t}(u,v)
	Define
	\begin{equation*}
		\widetilde A_t(u,v) := \int_0^1 \partial_x a\bigl( X_t(v)+\theta\Delta X_t,\mu_t \bigr)\,d\theta
	\end{equation*}
	and, for \(\ell=1,\dots,r\),
	\begin{equation*}
		\widetilde C_{\ell,t}(u,v) := \int_0^1 \partial_x\sigma_\ell\bigl( X_t(v)+\theta\Delta X_t \bigr)\,d\theta.
	\end{equation*}

	% the sde via fundamental theorem of calculus
	By the fundamental theorem of calculus,
	\begin{equation*}
		\begin{cases}
			d\Delta X_t = \widetilde A_t(u,v)\Delta X_t\,dt + \sum_{\ell=1}^{r} \widetilde C_{\ell,t}(u,v)\Delta X_t\,dB_t^\ell, \\[2mm]
			\Delta X_0=u-v.                                                                                                      \\[2mm]
		\end{cases}
	\end{equation*}

	% making sure the condition of lemma hold true
	Moreover, by \eqref{eq:basic-derivative-bounds}, for every
	\(z\in\R^d\),
	\begin{equation*}
		|\widetilde A_t(u,v)z| \le L_x|z|.
	\end{equation*}
	Consequently,
	\begin{equation*}
		\langle z,\widetilde A_t(u,v)z\rangle \le L_x|z|^2.
	\end{equation*}
	Jensen's inequality also gives, for every \(z\in\R^d\),
	\begin{align*}
		\sum_{\ell=1}^{r} |\widetilde C_{\ell,t}(u,v)z|^2
		 & \le \int_0^1 \sum_{\ell=1}^{r} \left| \partial_x\sigma_\ell\bigl( X_t(v)+\theta\Delta X_t \bigr)z \right|^2\,d\theta \\
		 & \le L_\sigma^2|z|^2.
	\end{align*}

	% Applying lemma for the liner sde moments
	Applying \Cref{lem:linear-sde-moment} with \(q=m/2\) and with \(q=m\), respectively, yields
	\begin{equation*}
		\E|X_t(u)-X_t(v)|^m \le C_me^{c_mt}|u-v|^m,
	\end{equation*}
	and
	\begin{equation}\label{eq:flow-increment-moment-2m}
		\E|X_t(u)-X_t(v)|^{2m} \le C_me^{c_mt}|u-v|^{2m}.
	\end{equation}

	% If \(u=v\), then \(\Delta X_t\equiv0\), so the desired estimate is trivial. Assume therefore that \(u\neq v\), and set
	% \begin{equation*}
	%     Z_t:=\frac{\Delta X_t}{|u-v|},
	%     \qquad
	%     \xi:=\frac{u-v}{|u-v|}.
	% \end{equation*}
	% Then \(|\xi|=1\), \(Z_0=\xi\), and \(Z_t\) solves
	% \begin{equation*}
	%     \begin{cases}
	%         dZ_t = \widetilde A_t(u,v)\,Z_t\,dt + \sum_{\ell=1}^{r} \widetilde C_{\ell,t}(u,v)\,Z_t\,dB_t^\ell, \\[2mm]
	%         Z_0 = \frac{u-v}{|u-v|}.                                                                   \\[2mm]
	%     \end{cases}
	% \end{equation*}
	% Applying \Cref{lem:linear-sde-moment} to this linear equation with exponent \(q=m/2\), we obtain
	% \begin{equation*}
	%     \E|Z_t|^m
	%     \le
	%     C_m e^{c_m t},
	% \end{equation*}
	% for constants \(C_m,c_m<\infty\) depending only on \(m,d,L,L_\sigma\).
	% Multiplying by \(|u-v|^m\) yields
	% \begin{equation}
	%     \E\,|X_t(u)-X_t(v)|^m
	%     \le
	%     C_m e^{c_m t}|u-v|^m,
	%     \qquad u,v\in\R^d.
	% \end{equation}

	\Cref{lem:linear-sde-moment} gives, for every \(q\ge1\) and every \(u\in\R^d\),
	\begin{equation*}
		\E\|J_t(u)\|^{2q} \le d^q\exp\left\{ q\bigl(2L_x+(2q-1)L_\sigma^2\bigr)t \right\}.
	\end{equation*}
	In particular, taking \(q=m/2\) and \(q=m\), we obtain
	\begin{equation}\label{eq:jacobian-moment}
		\E\|J_t(u)\|^m \le C_m e^{c_m t}, \qquad \E\|J_t(u)\|^{2m} \le C_m e^{c_m t},
	\end{equation}
	uniformly in \(u\in\R^d\).

	% define the difference D_t
	Fix \(u,v\in\R^d\), and write
	\begin{equation*}
		D_t:=J_t(u)-J_t(v).
	\end{equation*}
	% define the difference D_t
	Set
	\begin{equation*}
		A_t(w):=\partial_x a(X_t(w),\mu_t), \qquad C_{\ell,t}(w):=\partial_x\sigma_\ell(X_t(w)), \qquad \ell=1,\dots,r.
	\end{equation*}
	% set the sde for D_t equation
	Subtracting the Jacobian equations at \(u\) and \(v\), we obtain
	\begin{equation}\label{eq:jacobian-difference-sde}
		\begin{cases}
			dD_t = A_t(u)D_t\,dt + \sum_{\ell=1}^{r}C_{\ell,t}(u)D_t\,dB_t^\ell + F_t\,dt +
			\sum_{\ell=1}^{r}G_{\ell,t}\,dB_t^\ell, \\[1mm]
			D_0=0,                                  \\[1mm]
		\end{cases}
	\end{equation}
	where
	\begin{equation*}
		F_t := \bigl(A_t(u)-A_t(v)\bigr)J_t(v)
	\end{equation*}
	and
	\begin{equation*}
		G_{\ell,t} := \bigl(C_{\ell,t}(u)-C_{\ell,t}(v)\bigr)J_t(v), \qquad \ell=1,\dots,r.
	\end{equation*}

	% using the assumptions
	For every \(M\in\R^{d\times d}\),
	\begin{equation*}
		\langle M,A_t(u)M\rangle \le L_x\|M\|^2,
	\end{equation*}
	and
	\begin{equation*}
		\sum_{\ell=1}^{r} \|C_{\ell,t}(u)M\|^2 \le L_\sigma^2\|M\|^2.
	\end{equation*}
	Indeed, both inequalities follow by applying \eqref{eq:basic-derivative-bounds} to the columns of \(M\).

	Similarly, applying \eqref{eq:spatial-lipschitz-derivatives} columnwise gives
	\begin{equation*}
		\|F_t\| \le H|\Delta X_t|\,\|J_t(v)\|
	\end{equation*}
	and
	\begin{equation*}
		\left( \sum_{\ell=1}^{r}\|G_{\ell,t}\|^2 \right)^{1/2} \le H|\Delta X_t|\,\|J_t(v)\|.
	\end{equation*}
	Consequently, H\"older's inequality, \eqref{eq:flow-increment-moment-2m}, and \eqref{eq:jacobian-moment} imply
	\begin{align*}
		\E\|F_t\|^m + \E\left( \sum_{\ell=1}^{r}\|G_{\ell,t}\|^2 \right)^{m/2}
		 & \le C_m \left( \E|\Delta X_t|^{2m} \right)^{1/2} \left( \E\|J_t(v)\|^{2m} \right)^{1/2} \\
		 & \le C_me^{c_mt}|u-v|^m.
	\end{align*}

	Applying \Cref{lem:inhomogeneous-linear-moment} to \eqref{eq:jacobian-difference-sde}, and using \(D_0=0\), gives
	\begin{align*}
		\E\|D_t\|^m & \le C_m|u-v|^m\int_0^t e^{c_m(t-s)}e^{c_ms}\,ds \\
		            & \le C_me^{c_mt}|u-v|^m,
	\end{align*}
	after enlarging \(c_m\). Therefore,
	\begin{equation}\label{eq:jacobian-increment}
		\E\|J_t(u)-J_t(v)\|^m \le C_me^{c_mt}|u-v|^m, \qquad u,v\in\R^d.
	\end{equation}
	% In particular, the same estimate holds with the operator norm in place of the Hilbert--Schmidt norm.

	% define the cube???
	Let \(Q\subset\R^d\) be a compact cube containing \(K\), and choose \(u_0\in K\). By \eqref{eq:jacobian-increment}, \( u,v\in Q \)
	\begin{equation*}
		\E\|J_t(u)-J_t(v)\|^m \le C_me^{c_mt}|u-v|^m.
	\end{equation*}
	Since \(m>d\), this may be written as
	\begin{equation*}
		\E\|J_t(u)-J_t(v)\|^m \le C_me^{c_mt}|u-v|^{d+(m-d)}.
	\end{equation*}

	% application of Kolmogorov--Totoki theorem
	Hence the quantitative Kolmogorov--Totoki theorem, applied to the \(\R^{d\times d}\)-valued random field
	\begin{equation*}
		u\longmapsto J_t(u)-J_t(u_0),
	\end{equation*}
	gives
	\begin{equation}\label{eq:kolmogorov-totoki-bound}
		\E\left[\sup_{u\in Q} \|J_t(u)-J_t(u_0)\|^m \right] \le C_{m,Q}e^{c_mt}.
	\end{equation}
	Here we identify \(\R^{d\times d}\) with \(\R^{d^2}\).

	% % strange paragraph
	% Since \Cref{thm:jacobian-sde} already provides a spatially continuous version of \(J_t\), the continuous modification furnished by the
	% Kolmogorov--Totoki theorem agrees with this version on \(Q\) almost surely.

	% final triangle inequality
	For every \(u\in K\),
	\begin{equation*}
		\|J_t(u)\| \le \|J_t(u_0)\| + \|J_t(u)-J_t(u_0)\|.
	\end{equation*}
	Consequently,
	\begin{equation*}
		\mathcal K_t(K) \le \|J_t(u_0)\| + \sup_{u\in Q} \|J_t(u)-J_t(u_0)\|.
	\end{equation*}
	Using \eqref{eq:jacobian-moment}, \eqref{eq:kolmogorov-totoki-bound}, and \((a+b)^m\le 2^{m-1}(a^m+b^m)\), we obtain
	\begin{equation*}
		\E\mathcal K_t(K)^m \le C_{m,K}e^{c_mt}.
	\end{equation*}
	Finally,
	\begin{equation*}
		\E\mathcal K_t(K)^p\le \left( \E\mathcal K_t(K)^m \right)^{p/m} \le C_{p,K}e^{c_pt}.
	\end{equation*}
	This proves \eqref{eq:kt-exp-growth}.
\end{proof}

\section{The averaged Jacobian}\label{sec:averaged-jacobian}

The preceding pathwise estimates use the quantity \(\mathcal K_t(K)\) which controls the worst possible stretching of the flow on the
compact set \(K\). Consider instead the averaged Jacobian for the transported measure \(\mu_t=\mu_0\circ T_t^{-1}\),
\begin{equation*}
	\mathcal E_t := \int_{\R^d}\|J_t(u)\|^2\,\mu_0(du).
\end{equation*}
It measures average deformation of the flow over the initial mass, rather than
the maximal deformation on the whole support.

\begin{proposition}[Averaged Jacobian energy estimate]\label{prop:averaged-jacobian-energy}
	Assume that the flow is differentiable in the initial label and that there
	exist constants \(L_{\mathrm{sym}},\Gamma<\infty\) such that
	\begin{equation}\label{eq:one-sided-drift-jacobian}
		\langle z,\partial_x a(x,\mu)z\rangle \le L_{\mathrm{sym}}|z|^2, \qquad x,z\in\R^d,\ \mu\in\cP_2(\R^d),
	\end{equation}
	and
	\begin{equation}\label{eq:quadratic-diffusion-jacobian}
		\sum_{\ell=1}^{r}|\partial_x\sigma_\ell(x)z|^2 \le \Gamma^2|z|^2, \qquad x,z\in\R^d.
	\end{equation}
	Then
	\begin{equation}\label{eq:averaged-jacobian-bound}
		\E\mathcal E_t \le d\exp\{(2L_{\mathrm{sym}}+\Gamma^2)t\}.
	\end{equation}
\end{proposition}

\begin{proof}
	Recall that
	\begin{equation*}
		dJ_t(u) = A_t(u)J_t(u)\,dt + \sum_{\ell=1}^{r} C_{\ell,t}(u)J_t(u)\,dB_t^\ell,
	\end{equation*}
	where
	\begin{equation*}
		A_t(u)=\partial_x a(X_t(u),\mu_t), \qquad C_{\ell,t}(u)=\partial_x\sigma_\ell(X_t(u)).
	\end{equation*}
	Applying It\^o's formula to \(\|J_t(u)\|^2\) gives
	\begin{align*}
		d\|J_t(u)\|^2 & = \bigg[2\langle J_t(u),A_t(u)J_t(u)\rangle + \sum_{\ell=1}^{r}\|C_{\ell,t}(u)J_t(u)\|^2 \bigg]dt \\
		              & \quad + 2\sum_{\ell=1}^{r} \langle J_t(u),C_{\ell,t}(u)J_t(u)\rangle\,dB_t^\ell.
	\end{align*}
	Since
	\begin{equation*}
		\langle J,A_tJ\rangle = \sum_{i=1}^d\langle Je_i,A_tJe_i\rangle,
	\end{equation*}
	condition \eqref{eq:one-sided-drift-jacobian} gives
	\begin{equation*}
		2\langle J,A_tJ\rangle \le 2L_{\mathrm{sym}}\|J\|^2.
	\end{equation*}
	Similarly, \eqref{eq:quadratic-diffusion-jacobian} gives
	\begin{equation*}
		\sum_{\ell=1}^{r}\|C_{\ell,t}J\|^2 = \sum_{i=1}^d\sum_{\ell=1}^{r}|C_{\ell,t}Je_i|^2 \le \Gamma^2\|J\|^2.
	\end{equation*}
	Integrating with respect to \(\mu_0(du)\) and taking expectations removes the
	martingale term, hence
	\begin{equation*}
		\frac{d}{dt}\E\mathcal E_t \le (2L_{\mathrm{sym}}+\Gamma^2)\E\mathcal E_t.
	\end{equation*}
	Since \(J_0(u)=I\), we have \(\mathcal E_0=\|I\|^2=d\). Gronwall's
	lemma proves \eqref{eq:averaged-jacobian-bound}.
\end{proof}

To estimate \(\E[\mathcal K_t(K)^p]\), one must control increments \(J_t(u)-J_t(v)\), which requires assumptions on the spatial regularity
of \(\partial_x a\) and \(\partial_x\sigma\). The averaged quantity \(\mathcal E_t\) does not compare two labels. It only estimates each \(J_t(u)\)
and integrates in \(u\), so the one-sided bounds \eqref{eq:one-sided-drift-jacobian} and \eqref{eq:quadratic-diffusion-jacobian} are sufficient.

\section{Internal fluctuations}\label{sec:internal-fluctuations}

% Motivation for this section
This section concerns a separate kernel-type interacting model. In contrast to \eqref{eq:main-sdewi}, both the drift and diffusion may depend on
the order parameter \(\Theta_t\). Consequently, the assumptions of \cref{sec:jacobian-bound,sec:averaged-jacobian} are not imposed here.

% kernel-type interacting model
\begin{equation}\label{eq:kernel-sdewi}
	\begin{cases}
		dX_t(u) & = \bigg[ b(X_t(u),\Theta_t) + \int_{\R^d} F(X_t(u),X_t(v),\Theta_t)\,\mu_0(dv) \bigg]dt \\
		        & \quad+ \sum_{\ell=1}^r \sigma_\ell(X_t(u),\Theta_t)\,dB_t^\ell,                         \\[1mm]
		X_0(u)  & =u,\qquad u\in\R^d,                                                                     \\[1mm]
		\mu_t   & =\mu_0\circ X_t^{-1}.
	\end{cases}
\end{equation}

% the parameters of the kernel-type interacting model
Let \(q\ge1\), and let
\begin{equation*}
	\Phi:\R^d\to\R^q,\qquad b:\R^d\times\R^q\to\R^d,\qquad F:\R^d\times\R^d\times\R^q\to\R^d,
\end{equation*}
with \(\sigma_\ell:\R^d\times\R^q\to\R^d\), \(\ell=1,\dots,r\). Define
\begin{equation*}
	\theta(\mu):=\int_{\R^d}\Phi(x)\,\mu(dx), \qquad \Theta_t:=\theta(\mu_t)=\int_{\R^d}\Phi(X_t(v))\,\mu_0(dv).
\end{equation*}

% Dissipativity Assumptions on global Lipschitz and growth hypotheses.
\begin{assumption}[Well-posedness of \eqref{eq:kernel-sdewi}]\label{ass:kernel-wellposedness}
	The maps \(b,F,\sigma_\ell,\Phi\) are globally Lipschitz and have at most linear growth.
	More explicitly, there exists \(L<\infty\) such that
	\begin{align*}
		|\Phi(x)-\Phi(y)|           & \le L|x-y|,                               \\
		|b(x,\theta)-b(y,\eta)|     & \le L\big(|x-y|+|\theta-\eta|\big),       \\
		|F(x,z,\theta)-F(y,w,\eta)| & \le L\big(|x-y|+|z-w|+|\theta-\eta|\big),
	\end{align*}
	and
	\begin{equation*}
		\sum_{\ell=1}^r |\sigma_\ell(x,\theta)-\sigma_\ell(y,\eta)|^2 \le L^2\big(|x-y|^2+|\theta-\eta|^2\big).
	\end{equation*}
\end{assumption}

% Well-posedness of the SDEWI
Under \Cref{ass:kernel-wellposedness}, equation \eqref{eq:kernel-sdewi} admits a unique global continuous solution.

This section studies the pairwise energy for \eqref{eq:kernel-sdewi}
\begin{equation*}
	\mathcal D_t := \iint_{\R^d\times\R^d} |X_t(u)-X_t(v)|^2\,\mu_0(du)\mu_0(dv).
\end{equation*}

% ============================================================
% Dissipativity assumptions
% ============================================================
\subsection{Dissipativity assumptions}\label{subsec:dissipativity-assumptions}
Equivalently, if
\begin{equation*}
	\bar X_t:=\int_{\R^d}X_t(u)\,\mu_0(du),
\end{equation*}
and
\begin{equation*}
	\mathcal V_t := \int_{\R^d}|X_t(u)-\bar X_t|^2\,\mu_0(du),
\end{equation*}
then
\begin{equation*}
	\mathcal D_t=2\mathcal V_t.
\end{equation*}
Indeed,
\begin{align*}
	\mathcal D_t & = \iint |X_t(u)-X_t(v)|^2\,\mu_0(du)\mu_0(dv)                                                \\
	             & = \iint \big( |X_t(u)|^2+|X_t(v)|^2 -2\langle X_t(u),X_t(v)\rangle \big)\,\mu_0(du)\mu_0(dv) \\
	             & = 2\int |X_t(u)|^2\,\mu_0(du)- 2\left|\int X_t(u)\,\mu_0(du)\right|^2                        \\
	             & = 2\int |X_t(u)-\bar X_t|^2\,\mu_0(du)= 2\mathcal V_t.
\end{align*}
We assume the following dissipativity condition.
\begin{assumption}[Pairwise dissipativity]\label{ass:pairwise-dissipativity}
	There exists \(\lambda>0\) such that for every \(x,y\in\R^d\), every \(\theta\in\R^q\), and every
	\(\nu\in\cP_2(\R^d)\),
	\begin{align*}
		 & 2\left\langle x-y, b(x,\theta)-b(y,\theta) + \int_{\R^d} \big[F(x,z,\theta)-F(y,z,\theta)\big] \nu(dz) \right\rangle \\
		 & \qquad + \sum_{\ell=1}^r |\sigma_\ell(x,\theta)-\sigma_\ell(y,\theta)|^2 \le -\lambda |x-y|^2.
	\end{align*}
\end{assumption}

The dissipativity condition covers several basic models.

\begin{example}[Ornstein--Uhlenbeck SDEWI]
	Consider
	\begin{equation*}
		dX_t(u)= -\lambda_0\big(X_t(u)-m_t\big)\,dt + \sigma(X_t(u))\,dB_t, \qquad m_t=\int_{\R^d} X_t(v)\,\mu_0(dv).
	\end{equation*}
	If
	\begin{equation*}
		\|\sigma(x)-\sigma(y)\| \le L_\sigma |x-y|,
	\end{equation*}
	then
	\begin{equation*}
		\E\mathcal D_t \le e^{-(2\lambda_0-L_\sigma^2)t}\mathcal D_0.
	\end{equation*}
	Thus the cloud contracts whenever \(2\lambda_0>L_\sigma^2\).
\end{example}

\begin{example}[Gradient systems]
	Take
	\begin{equation*}
		b(x,\theta)=-\partial V(x), \qquad F(x,y,\theta)=-\partial W(x-y).
	\end{equation*}
	Then
	\begin{equation*}
		dX_t(u) =  \left[ -\partial V(X_t(u)) - \int_{\R^d} \partial W(X_t(u)-X_t(v))\,\mu_0(dv) \right]dt + \sigma(X_t(u))\,dB_t.
	\end{equation*}
	If
	\begin{equation*}
		\partial_x^2 V\ge \rho I, \qquad \partial_x^2 W\ge \rho_W I, \qquad \|\sigma(x)-\sigma(y)\| \le L_\sigma |x-y|,
	\end{equation*}
	then
	\begin{equation*}
		\E\mathcal D_t \le e^{-(2\rho+2\rho_W-L_\sigma^2)t}\mathcal D_0.
	\end{equation*}
\end{example}

\subsection{Proof}\label{subsec:synchronization-proof}
\begin{theorem}\label{thm:synchronization-dissipativity}
	Assume that \(\mu_0\in\cP_2(\R^d)\) and that \Cref{ass:pairwise-dissipativity} holds. Then
	\begin{equation*}
		\E\mathcal D_t \le e^{-\lambda t}\mathcal D_0.
	\end{equation*}

	In particular,
	\begin{equation*}
		\E W_2^2(\mu_t,\delta_{\bar X_t}) \le e^{-\lambda t}W_2^2(\mu_0,\delta_{\bar X_0}).
	\end{equation*}
	Thus the transported measure \(\mu_t\) concentrates around its random center of mass \(\bar X_t\).
\end{theorem}

% Proof idea:
% 1. Differentiate the stochastic flow.
\begin{proof}
	% Proof roadmap: first prove pairwise contraction, then integrate over labels.
	Fix two labels \(u,v\in\R^d\). Define the difference
	\begin{equation*}
		Z_t^{u,v}:=X_t(u)-X_t(v).
	\end{equation*}
	We first prove that under \Cref{ass:pairwise-dissipativity}
	\begin{equation*}
		\E|Z_t^{u,v}|^2 \le e^{-\lambda t}|u-v|^2,
	\end{equation*}
	then we integrate this estimate with respect to \(\mu_0(du)\mu_0(dv)\).

	Writing \eqref{eq:kernel-sdewi} for \(X_t(u)\) and for \(X_t(v)\) and then subtracting the two equations gives
	\begin{equation*}
		dZ_t^{u,v} = G_t^{u,v}\,dt + \sum_{\ell=1}^r H_{\ell,t}^{u,v}\,dB_t^\ell,
	\end{equation*}
	where the drift coefficient
	\begin{align*}
		G_t^{u,v} & := b(X_t(u),\Theta_t)-b(X_t(v),\Theta_t)                                                    \\
		          & +  \int_{\R^d} \big[ F(X_t(u),X_t(w),\Theta_t) - F(X_t(v),X_t(w),\Theta_t)\big]\,\mu_0(dw),
	\end{align*}
	and the diffusion coefficient is
	\begin{equation*}
		H_{\ell,t}^{u,v} := \sigma_\ell(X_t(u),\Theta_t) - \sigma_\ell(X_t(v),\Theta_t).
	\end{equation*}

	Since \(\mu_t =\mu_0\circ X_t^{-1}\), we may rewrite the interaction term as
	\begin{equation*}
		\int_{\R^d}F(X_t(u),X_t(w),\Theta_t)\,\mu_0(dw) = \int_{\R^d}F(X_t(u),z,\Theta_t)\,\mu_t(dz).
	\end{equation*}
	Thus
	\begin{align*}
		G_t^{u,v} & := b(X_t(u),\Theta_t)-b(X_t(v),\Theta_t)                                          \\
		          & + \int_{\R^d} \big[ F(X_t(u),z,\Theta_t) - F(X_t(v),z,\Theta_t) \big]\,\mu_t(dz).
	\end{align*}

	Applying It\^o's formula to \(|Z_t^{u,v}|^2\), we obtain
	\begin{align*}
		d|Z_t^{u,v}|^2 & = 2\langle Z_t^{u,v},G_t^{u,v}\rangle\,dt + \sum_{\ell=1}^r |H_{\ell,t}^{u,v}|^2\,dt \\
		               & \quad + 2\sum_{\ell=1}^r \langle Z_t^{u,v},H_{\ell,t}^{u,v}\rangle\,dB_t^\ell.
	\end{align*}
	Apply \Cref{ass:pairwise-dissipativity} with
	\(x=X_t(u)\), \(y=X_t(v)\), \(\theta=\Theta_t\), and \(\nu=\mu_t\). Then
	\begin{equation*}
		2\langle Z_t^{u,v},G_t^{u,v}\rangle + \sum_{\ell=1}^r |H_{\ell,t}^{u,v}|^2 \le -\lambda |Z_t^{u,v}|^2.
	\end{equation*}
	Therefore
	\begin{equation}\label{eq:pairwise-contraction-differential}
		d|Z_t^{u,v}|^2 \le -\lambda |Z_t^{u,v}|^2\,dt + 2\sum_{\ell=1}^r \langle Z_t^{u,v},H_{\ell,t}^{u,v}\rangle\,dB_t^\ell.
	\end{equation}

	Set the new variable
	\begin{equation*}
		Y_t^{u,v}:=e^{\lambda t}|Z_t^{u,v}|^2.
	\end{equation*}
	By It\^o's formula,
	\begin{align*}
		dY_t^{u,v} & = e^{\lambda t} \left[ \lambda |Z_t^{u,v}|^2 + 2\langle Z_t^{u,v},G_t^{u,v}\rangle \right.  \\
		           & \quad \left. + \sum_{\ell=1}^r |H_{\ell,t}^{u,v}|^2 \right]dt                               \\
		           & \quad+ 2e^{\lambda t} \sum_{\ell=1}^r \langle Z_t^{u,v},H_{\ell,t}^{u,v}\rangle\,dB_t^\ell.
	\end{align*}
	By the drift bound \eqref{eq:pairwise-contraction-differential}, the drift term above is nonpositive. Hence
	\begin{equation*}
		dY_t^{u,v} \le 2e^{\lambda t} \sum_{\ell=1}^r \langle Z_t^{u,v},H_{\ell,t}^{u,v}\rangle\,dB_t^\ell.
	\end{equation*}
	% Thus \(Y_t^{u,v}\) is a nonnegative local supermartingale.

	To make this completely rigorous, introduce the localization sequence
	\begin{equation*}
		\tau_R := \inf\{t\ge0: |X_t(u)|+|X_t(v)|+|\Theta_t|\ge R\}.
	\end{equation*}
	Since the solution is global and continuous, \(\tau_R\uparrow\infty\) almost surely as \(R\to\infty\). Stopping this inequality at
	\(\tau_R\), we get
	\begin{align*}
		Y_{t\wedge\tau_R}^{u,v} & \le Y_0^{u,v}                                                                                                       \\
		                        & \quad + 2\sum_{\ell=1}^r \int_0^{t\wedge\tau_R} e^{\lambda s} \langle Z_s^{u,v},H_{\ell,s}^{u,v}\rangle\,dB_s^\ell.
	\end{align*}
	The stopped stochastic integral is a true martingale. Therefore, taking
	expectations gives
	\begin{equation*}
		\E Y_{t\wedge\tau_R}^{u,v} \le Y_0^{u,v}.
	\end{equation*}
	Since \(Y_0^{u,v}=|u-v|^2\), we have
	\begin{equation*}
		\E\left[ e^{\lambda(t\wedge\tau_R)} |Z_{t\wedge\tau_R}^{u,v}|^2 \right] \le |u-v|^2.
	\end{equation*}
	Letting \(R\to\infty\) and using Fatou's lemma yields
	\begin{equation*}
		\E\left[e^{\lambda t}|Z_t^{u,v}|^2\right] \le |u-v|^2.
	\end{equation*}
	Therefore
	\begin{equation*}
		\E|X_t(u)-X_t(v)|^2 \le  e^{-\lambda t}|u-v|^2.
	\end{equation*}
	This proves the pairwise synchronization estimate.

	Now integrate the pairwise synchronization estimate with respect to \(\mu_0(du)\mu_0(dv)\). Since \(\mu_0\in\cP_2(\R^d)\),
	\begin{equation*}
		\mathcal D_0 = \iint |u-v|^2\,\mu_0(du)\mu_0(dv)<\infty.
	\end{equation*}
	By Fubini's theorem,
	\begin{align*}
		\E\mathcal D_t & = \E \iint |X_t(u)-X_t(v)|^2\,\mu_0(du)\mu_0(dv)      \\
		               & = \iint \E|X_t(u)-X_t(v)|^2\,\mu_0(du)\mu_0(dv)       \\
		               & \le e^{-\lambda t} \iint |u-v|^2\,\mu_0(du)\mu_0(dv).
	\end{align*}
	Hence \(\E\mathcal D_t\le e^{-\lambda t}\mathcal D_0\), proving
	the first estimate in the theorem.

	Therefore
	\begin{equation*}
		\E\mathcal V_t = \frac12\E\mathcal D_t \le \frac12 e^{-\lambda t}\mathcal D_0 = e^{-\lambda t}\mathcal V_0.
	\end{equation*}
	This proves the internal fluctuation decay estimate.

	Finally, since \(\mu_t =\mu_0\circ X_t^{-1}\),
	\begin{equation*}
		\int_{\R^d}|x-\bar X_t|^2\,\mu_t(dx) = \int_{\R^d}|X_t(u)-\bar X_t|^2\,\mu_0(du) = \mathcal V_t.
	\end{equation*}
	For the Dirac measure \(\delta_{\bar X_t}\),
	\begin{equation*}
		W_2^2(\mu_t,\delta_{\bar X_t}) = \int_{\R^d}|x-\bar X_t|^2\,\mu_t(dx).
	\end{equation*}
	Therefore
	\begin{equation*}
		W_2^2(\mu_t,\delta_{\bar X_t})=\mathcal V_t.
	\end{equation*}
	Taking expectations and using the previous estimate gives
	\begin{equation*}
		\E W_2^2(\mu_t,\delta_{\bar X_t}) \le e^{-\lambda t}W_2^2(\mu_0,\delta_{\bar X_0}).
	\end{equation*}
\end{proof}
